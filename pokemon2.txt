"""

# Types
type Episode {
  id: ID!
  number: Int!
  title: String!
  titleJapanese: String
  series: Series!
  season: Int!
  airDate: String!
  airDateJapanese: String
  synopsis: String!
  type: EpisodeType!
  duration: Int!
  characters: [Character!]!
  featuredPokemon: [AnimePokemon!]!
  location: Location
  badges: [Badge!]
  captures: [Capture!]
  evolutions: [Evolution!]
  battles: [Battle!]
  imageUrl: String
  watchUrl: String
}

type Series {
  id: ID!
  name: SeriesName!
  displayName: String!
  season: Int!
  episodes: [Episode!]!
  startDate: String!
  endDate: String
  region: String!
  mainCharacters: [Character!]!
  totalEpisodes: Int!
  description: String
  posterUrl: String
}

type Character {
  id: ID!
  name: String!
  nameJapanese: String
  role: CharacterRole!
  age: Int
  hometown: String
  debut: Episode!
  biography: String!
  team: [AnimePokemon!]!
  badges: [Badge!]
  achievements: [Achievement!]
  appearances: [Episode!]!
  voiceActor: VoiceActor
  imageUrl: String
}

  imageUrl: String

type VoiceActor {

# Input Types
input EpisodeFilterInput {
  series: SeriesName
  season: Int
  type: EpisodeType
  character: String
  pokemon: String
  hasGymBattle: Boolean
  hasCapture: Boolean
  hasEvolution: Boolean
  searchTerm: String
}

input CharacterFilterInput {
  role: CharacterRole
  series: SeriesName
  region: String
  hasPokemon: String
  searchTerm: String
}

input PokemonFilterInput {
  types: [PokemonType!]
  owner: String
  region: String
  isReleased: Boolean
  searchTerm: String
}

input DateRangeInput {
  startDate: String!
  endDate: String!
}

# Pagination
type EpisodeConnection {
  edges: [EpisodeEdge!]!
  pageInfo: PageInfo!
  totalCount: Int!
}

type EpisodeEdge {
  node: Episode!
  cursor: String!
}

type PageInfo {
  hasNextPage: Boolean!
  hasPreviousPage: Boolean!
  startCursor: String
  endCursor: String
}

# Root Types
type Query {
  # Episode queries
  episode(id: ID!): Episode
  episodeByNumber(number: Int!, series: SeriesName!): Episode
  episodes(
    filter: EpisodeFilterInput
    first: Int
    after: String
  ): EpisodeConnection!
  
  # Series queries
  series(name: SeriesName!): Series
  allSeries: [Series!]!
  currentSeries: Series!
  
  # Character queries
  character(id: ID!): Character
  characterByName(name: String!): Character
  characters(filter: CharacterFilterInput): [Character!]!
  
  # Pokemon queries
  pokemon(id: ID!): AnimePokemon
  allPokemon(filter: PokemonFilterInput): [AnimePokemon!]!
  pokemonByOwner(ownerId: ID!): [AnimePokemon!]!
  
  # Ash's journey
  ashKetchum: Character!
  ashsPokemon: [AnimePokemon!]!
  ashsBadges: [Badge!]!
  
  # Team Rocket
  teamRocket: TeamRocket!
  
  # Battle queries
  battle(id: ID!): Battle
  battles(characterId: ID): [Battle!]!
  gymBattles(region: String): [Battle!]!
  
  # Movie queries
  movie(id: ID!): Movie
  movieByNumber(number: Int!): Movie
  allMovies: [Movie!]!
  
  # Location queries
  location(id: ID!): Location
  locationsByRegion(region: String!): [Location!]!
  
  # Voice actor queries
  voiceActor(id: ID!): VoiceActor
  voiceActorsByLanguage(language: String!): [VoiceActor!]!
  
  # Stats
  animeStats: AnimeStats!
}

type AnimeStats {
  totalEpisodes: Int!
  totalSeries: Int!
  totalCharacters: Int!
  totalPokemon: Int!
  totalMovies: Int!
  regions: [String!]!
}

type Mutation {
  # Watchlist mutations
  addToWatchlist(userId: ID!, episodeId: ID!): Boolean!
  removeFromWatchlist(userId: ID!, episodeId: ID!): Boolean!
  markAsWatched(userId: ID!, episodeId: ID!): Boolean!
  
  # Rating mutations
  rateEpisode(episodeId: ID!, rating: Int!): Episode!
  rateMovie(movieId: ID!, rating: Int!): Movie!
  
  # Favorite mutations
  addFavoriteCharacter(userId: ID!, characterId: ID!): Character!
  addFavoritePokemon(userId: ID!, pokemonId: ID!): AnimePokemon!
}

type Subscription {
  newEpisodeReleased: Episode!
  newMovieReleased: Movie!
}

schema {
  query: Query
  mutation: Mutation
  subscription: Subscription
}
